# ICON

Usage:

```javascript
import { Facebook } from 'alien-ui/icons';

render() {
  return (
    <Facebook width="25" height="25" fill="#4172B8" />
  );
}
```

# ICON WITH HOVER FX

Usage:

```javascript
import { IconHover } from 'alien-ui';
import { Facebook } from 'alien-ui/icons';

render() {
  return (
    <IconHover hover="lightgrey">
      <Facebook width="25" height="25" fill="#4172B8" />
    </IconHover>
  );
}
```
